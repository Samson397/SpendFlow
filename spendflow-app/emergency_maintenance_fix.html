Emergency Maintenance Mode Fix

Copy and paste this code into your browser console:

// Emergency Maintenance Mode Disable
(async () => {
  try {
    // Import Firebase modules
    const { doc, updateDoc } = await import("https://www.gstatic.com/firebasejs/9.22.0/firebase-firestore.js");
    
    // You'll need to replace these with your actual Firebase config
    const firebaseConfig = {
      // Copy your Firebase config here
    };
    
    // Initialize Firebase (you'll need to set this up)
    const app = initializeApp(firebaseConfig);
    const db = getFirestore(app);
    
    // Disable maintenance mode
    const settingsRef = doc(db, "settings", "app");
    await updateDoc(settingsRef, { 
      maintenanceMode: false,
      updatedAt: new Date().toISOString() 
    });
    
    console.log("✅ Maintenance mode disabled!");
    alert("Maintenance mode disabled! Refresh the page.");
    
  } catch (error) {
    console.error("❌ Error:", error);
    alert("Error disabling maintenance mode. Check console for details.");
  }
})();

Alternative: Firebase Console

  1. Go to Firebase Console
  2. Select your project
  3. Go to Firestore Database
  4. Navigate to: settings/app document
  5. Set maintenanceMode field to: false
  6. Or delete the maintenanceMode field
